# sitewomen/women/converters
class FourDigitYearConverter:
    # Регулярное выражение
    regex = "[0-9]{4}"

    # Преобразовывает фрагмент url в требуемый тип данных
    def to_python(self, value):
        return int(value)

    # Наоборот преобразуем данные в нужный формат для создания URL
    def to_url(self, value):
        return "%04d" % value
